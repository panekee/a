const int led = 13;
const int dot = 250;

void setup() {
  pinMode(led, OUTPUT);
}

void blinkDot() {
  digitalWrite(led, HIGH); delay(dot);
  digitalWrite(led, LOW); delay(dot);
}

void blinkDash() {
  digitalWrite(led, HIGH); delay(dot*3);
  digitalWrite(led, LOW); delay(dot);
}

void loop() {
  blinkDot(); blinkDash();
  delay(dot*7);
}
